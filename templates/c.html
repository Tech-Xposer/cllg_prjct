#include<stdio.h>

void add(float a, float b)
{
    float result;
    result=a+b;
    printf("Result: %.2f",result);
}
void sub(float a, float b)
{
    float result;
    result=a-b;
    printf("Result:%.2f",result);
}
void mul(float a, float b)
{
    float result;
    result=a*b;
    printf("Result:%.2f",result);
}
void div(float a, float b)
{
    float result;
    result=a/b;
    printf("Result:%.2f",result);
}

int main()
{
    float num1,num2;
    char op;
    printf("Enter first number:");
    scanf("%f",&num1);
    printf("Enter the operation:");
    fflush(stdin);
    scanf(" %c",&op);
    printf("Enter the second numner:");
    scanf("%f",&num2);
    switch (op)
    {
    case '+':
        add(num1,num2);
        break;
    case '-':
        sub(num1,num2);
        break;
    case '*':
        mul(num1,num2);
        break;
    case '/':
        div(num1,num2);
        break;
    default:
        printf("INVALID OPERATION....Enter the valid operation!");
        break;
    }
}
